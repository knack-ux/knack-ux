  <Story name="Basic">
    {() => {
      const [checked, setChecked] = useState([]);
      return (
        <CheckboxGroup
          id="icecream"
          label="Which kinds of ice cream do you find nice?"
          checked={checked}
          options={[
            { key: 'Vanilla', value: 'vanilla' },
            { key: 'Chocolate', value: 'choco' },
            { key: 'Strawberry', value: 'strawberry' }
          ]}
          onChange={(event) => setChecked(event.target.value)}
        />
      )
    }}
